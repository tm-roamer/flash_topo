当完成上一小节的初始化配置后,进行下面的全局设置

	//---------------------------------配置绘画---------------------------------------------//
	var draw_options = {
	   //network: network_xml,	//显示数据,json,xml都可以(下面小节,详细讲解)
	   nodeLabelsVisible:true,  //是否显示线的标签
	   edgeLabelsVisible:true,  //是否显示点的标签
	   /*layout: { //"CompoundSpringEmbedder",//layout: "Tree",Circle,ForceDirected,Radial,CompoundSpringEmbedder
		 name:"Preset",
		 options:{
			fitToScreen :false, //false左上角为0,0 true为节点居中,自动适应屏幕,0,0不是中心点
			points: null//pointData
		 }
	   },  */
	   layout:"ForceDirected",		//设置布局(下面小节,详细讲解)
	   visualStyle: visual_style,	//设置样式(下面小节,详细讲解)
	   panZoomControlVisible:true,  //是否显示控制台
	   //panZoomControlPosition:"bottomRight", //显示位置topLeft,topCenter,topRight,middleLeft,middleCenter,middleRight,bottomLeft,bottomCenter,bottomRight(默认)
	   nodeTooltipsEnabled: true,	//是否显示点的tooltip
	   edgeTooltipsEnabled: true	//是否显示线的tooltip
		//preloadImages: true,		//渲染之前是否先加载图片
	    //edgesMerged:true			//是否显示线的箭头
	};
	//---------------------------------开始绘画---------------------------------------------//
	//第一次绘制加载样式和配置
	vis.draw(draw_options);
